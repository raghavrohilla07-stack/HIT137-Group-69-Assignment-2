import turtle

def draw_edge(length, depth):
    """Recursive function to draw one edge with indentations."""
    if depth == 0:
        turtle.forward(length)
    else:
        length /= 3
        # First segment
        draw_edge(length, depth - 1)
        # Turn left 60° to make equilateral triangle
        turtle.left(60)
        draw_edge(length, depth - 1)
        # Turn right 120° to complete indentation
        turtle.right(120)
        draw_edge(length, depth - 1)
        # Turn left 60° to return to original direction
        turtle.left(60)
        draw_edge(length, depth - 1)


def draw_polygon(sides, length, depth):
    """Draw the recursive polygon pattern."""
    for _ in range(sides):
        draw_edge(length, depth)
        turtle.right(360 / sides)  # turn to next side


def main():
    # User input
    sides = int(input("Enter the number of sides: "))
    length = int(input("Enter the side length: "))
    depth = int(input("Enter the recursion depth: "))

    # Setup turtle
    turtle.speed(0)   # fastest speed
    turtle.hideturtle()
    turtle.penup()
    turtle.goto(-length // 2, length // 3)  # shift drawing for better view
    turtle.pendown()

    # Draw pattern
    draw_polygon(sides, length, depth)

    turtle.done()


if __name__ == "__main__":
    main()
